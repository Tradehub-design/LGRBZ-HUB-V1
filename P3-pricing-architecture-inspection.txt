   234	/**
   235	 * Combines:
   236	 *
   237	 * - current provider results;
   238	 * - the previous render's valid quote map;
   239	 * - browser-persisted valid quotes.
   240	 *
   241	 * Invalid refresh responses never enter this function because the adapter
   242	 * rejects them first.
   243	 */
   244	export function mergeResilientPortfolioQuotes(input: {
   245	  current:
   246	    Readonly<
   247	      Record<string, QuoteSnapshot>
   248	    >;
   249	
   250	  previous:
   251	    Readonly<
   252	      Record<string, QuoteSnapshot>
   253	    >;
   254	
   255	  persisted:
   256	    Readonly<
   257	      Record<string, QuoteSnapshot>
   258	    >;
   259	
   260	  securityIds:
   261	    readonly string[];
   262	
   263	  now?: string;
   264	}): Record<string, QuoteSnapshot> {
   265	  const now =
   266	    input.now ??
   267	    new Date().toISOString();
   268	
   269	  const nowTimestamp =
   270	    Date.parse(now);
   271	
   272	  const safeNow =
   273	    Number.isFinite(nowTimestamp)
   274	      ? nowTimestamp
   275	      : Date.now();
   276	
   277	  const result:
   278	    Record<string, QuoteSnapshot> = {};
   279	
   280	  for (
   281	    const securityId of
   282	    input.securityIds
   283	  ) {
   284	    const current =
   285	      input.current[securityId];
   286	
   287	    const previous =
   288	      input.previous[securityId];
   289	
   290	    const persisted =
   291	      input.persisted[securityId];
   292	
   293	    const retainedCandidates =
   294	      [
   295	        previous,
   296	        persisted,
   297	      ]
   298	        .filter(
   299	          (
   300	            quote,
   301	          ): quote is QuoteSnapshot =>
   302	            Boolean(quote) &&
   303	            validPositivePrice(
   304	              quote,
   305	            ) &&
   306	            withinRetentionAge(
   307	              quote,
   308	              safeNow,
   309	            ),
   310	        )
   311	        .map((quote) =>
   312	          retainedQuote(
   313	            quote,
   314	            now,
   315	          ),
   316	        );
   317	
   318	    const strongestRetained =
   319	      selectStrongestPortfolioQuote(
   320	        retainedCandidates,
   321	      );
   322	
   323	    if (
   324	      current &&
   325	      validPositivePrice(current)
   326	    ) {
   327	      if (!strongestRetained) {
   328	        result[securityId] =
   329	          current;
   330	
   331	        continue;
   332	      }
   333	
   334	      /**
   335	       * A fresh provider quote always replaces retained data. A degraded,
   336	       * stale or previous-close response replaces retained data only when it
   337	       * has at least equivalent canonical strength.
   338	       */
   339	      if (
   340	        current.source === "LIVE" &&
   341	        (
   342	          current.quality === "LIVE" ||
   343	          current.quality === "DELAYED"
   344	        )
   345	      ) {
   346	        result[securityId] =
   347	          current;
   348	
   349	        continue;
   350	      }
   351	
   352	      result[securityId] =
   353	        canonicalQuotePriority(current) >=
   354	        canonicalQuotePriority(
   355	          strongestRetained,
   356	        )
   357	          ? current
   358	          : strongestRetained;
   359	
   360	      continue;
   361	    }
   362	
   363	    if (strongestRetained) {
   364	      result[securityId] =
   365	        strongestRetained;
   366	    }
   367	  }
   368	
   369	  return result;
   370	}
